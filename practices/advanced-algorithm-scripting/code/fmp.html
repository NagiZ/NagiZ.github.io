var firstMissingPositive = function(nums) {
		    var arr = [];
		    for (var i = 0; i < nums.length; i++) {
		        const num = nums[i];
		        if (num > 0 ) {
		            arr[num - 1] = 1;
		        }
		    }
		    for (var i = 0; i < arr.length; i++) {
		        if (!arr[i]) {
		            return i + 1;
		        }
		    }
		    return arr.length + 1;
		};

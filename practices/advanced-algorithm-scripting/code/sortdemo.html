function quickSort(arr){
  if (arr.length<=1) {
    return arr;
  }
  var arrL = [],
    arrR = [],
      num = Math.floor(arr.length/2);
  var mid = [arr[num]];
  for(var i = 0; i < arr.length; i++){
    if (arr[i] < mid[0]) {
      arrL.push(arr[i]);
    }else if (arr[i] > mid[0]) {
      arrR.push(arr[i])
    }
  }
  return quickSort(arrL).concat(mid.concat(quickSort(arrR)))
}
